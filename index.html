    function connectNativeMQTT() {
        const url = `wss://${BROKER_HOST}:${WEBSOCKET_PORT}/mqtt`;
        ws = new WebSocket(url, ["mqtt"]);
        ws.binaryType = "arraybuffer";

        ws.onopen = () => {
            sendMqttConnectPacket();
        };

        ws.onclose = () => {
            clearInterval(pingTimer);
            statusBar.classList.remove('connected');
            statusDot.classList.remove('connected');
            statusText.innerText = "Status: Reconnecting to HiveMQ Broker...";
            setTimeout(connectNativeMQTT, 4000);
        };

        ws.onmessage = (event) => {
            const payload = new Uint8Array(event.data);
            if (payload.length === 0) return;

            const packetType = payload[0] >> 4;

            if (packetType === 2) { // CONNACK Authorization Success Handler
                statusBar.classList.add('connected');
                statusDot.classList.add('connected');
                statusText.innerText = `Connected: Listening live on [${TOPIC_SUBSCRIBE}]`;
                appendLine("SYSTEM LOG: Authorization approved! Pipeline active.", "sys-log");
                sendMqttSubscribePacket();
                
                clearInterval(pingTimer);
                pingTimer = setInterval(() => {
                    if (ws.readyState === 1) ws.send(new Uint8Array([0xC0, 0x00])); // Send PINGREQ
                }, 15000);
            }

            if (packetType === 3) { // PUBLISH Payload Frame Router
                let idx = 1;
                let remLen = 0;
                let multiplier = 1;
                do {
                    remLen += (payload[idx] & 127) * multiplier;
                    multiplier *= 128;
                } while ((payload[idx++] & 128) !== 0);

                const topicLen = (payload[idx++] << 8) + payload[idx++];
                idx += topicLen;

                // Strip the application wrapper array layer and route payload downstream
                const rawDataBytes = payload.subarray(idx);
                parseMegmeetCorePacket(rawDataBytes);
            }
        };
    }

    function sendMqttConnectPacket() {
        const cid = "WebHub" + Math.floor(Math.random() * 1000);
        const enc = new TextEncoder();
        const bUser = enc.encode(MQTT_USER);
        const bPass = enc.encode(MQTT_PASS);
        const bCid = enc.encode(cid);
        const remLen = 12 + (2 + bCid.length) + (2 + bUser.length) + (2 + bPass.length);
        const pkt = new Uint8Array(2 + remLen);

        pkt[idx++] = 0;
        pkt[idx++] = 4;

        pkt[idx++] = 194;

        pkt[idx++] = 60;
        pkt[idx++] = (bCid.length >> 8);
        pkt[idx++] = (bCid.length & 255);

        idx += bCid.length;

// --- RENDERING BITWISE SIDEBAR SELECTIONS ---
function renderCheckboxBitmaps(rawDataBytes) {
    BIT_INDICATOR_MAPS.forEach(indicator => {
        const byteVal = rawDataBytes[indicator.byteIndex];
        const isSet = (byteVal & indicator.bitMask) !== 0;
        const el = document.getElementById(indicator.domId);
        
        if (el) {
            if (isSet) {
                el.classList.add('active');
            } else {
                el.classList.remove('active');
            }
        }
    });
}

// --- TERMINAL CONSOLE MATRIX COLUMN PIPELINE BUILDER ---
function generateTelemetryLogBlock(rawDataBytes, compTime) {
    const timeStr = new Date().toLocaleTimeString();
    
    // 1. Create Hexadecimal View String
    const hexColumn = Array.from(rawDataBytes)
        .map(b => b.toString(16).toUpperCase().padStart(2, '0'))
        .join(' ');
    
    // 2. Create Readable Print ASCII String View Column
    const asciiColumn = Array.from(rawDataBytes)
        .map(b => (b >= 32 && b <= 126) ? String.fromCharCode(b) : '.')
        .join('');

    // 3. Assemble Log Block Structure
    return `[${timeStr}] \n` +
           `Comp Uptime Status: ${compTime !== null ? compTime + ' Sec' : '--'}\n` +
           `HEX PAYLOAD STREAM : ${hexColumn}\n` +
           `ASCII DATA EXTRACT : ${asciiColumn}\n` +
           `--------------------------------------------------------------------------------`;
}
